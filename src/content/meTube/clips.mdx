import LiteYouTube from '../../components/LiteYouTube.astro';
import { videos } from './clips-data.ts';

export const grouped = videos.reduce((acc, v) => {
  (acc[v.group] ??= []).push(v);
  return acc;
}, {});

{/* First display all my clips */}

<section class="mb-12 not-prose">
    <h2 class="text-2xl font-bold mb-6 pb-2 border-b border-zinc-700 text-white">Clips</h2>
    <div class="video-grid">
        {grouped["my-clips"].map(clip => (
        <div class="card">
            <div class="video-wrapper">
                <LiteYouTube videoId={clip.id} title={clip.title} pillarbox={clip.pillarbox ?? false} />
            </div>
            <div class="video-info">
                <div class="video-title">{clip.title}</div>
            </div>
        </div>
        ))}
    </div>
</section>

{/* Then my short format videos */}

<section class="mb-12 not-prose">
    <h2 class="text-2xl font-bold mb-6 pb-2 border-b border-zinc-700 text-white">Shorts</h2>
    <div class="shorts-grid">
    {grouped["shorts"].map(clip => (
        <div class="card">
            <div class="video-wrapper shorts">
                {clip.type === "mp4" ? 
                    <video 
                        controls 
                        preload="metadata" 
                        class="absolute top-0 left-0 w-full h-full"
                        style="object-fit: contain;"
                    >
                    <source src={clip.src} type="video/mp4" />
                    </video>
                : <LiteYouTube videoId={clip.id} title={clip.title} />
                }
            </div>
            <div class="video-info">
            <div class="video-title">{clip.title}</div>
            </div>
        </div>
        ))}
    </div>    
</section>

{/* Lastly externally sourced clips */}

<section class="mb-12 not-prose">
    <h2 class="text-2xl font-bold mb-6 pb-2 border-b border-zinc-700 text-white">de WWW</h2>
    <div class="video-grid">
        {grouped["ext-clips"].map(ext => (
            <div class="card">
                <div class="video-wrapper">
                    <LiteYouTube videoId={ext.id} title={ext.title} params={ext.params} />
                </div>
                <div class="video-info">
                    <div class="video-title">{ext.title}</div>
                </div>
            </div>
        ))}    
    </div>
</section>
